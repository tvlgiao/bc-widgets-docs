    Credit Promotion System — Implementation Plan¶

    For Claude: REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

    Goal: Grant one-time free credit to all users on app load, with admin banner and chatbot inline notice.

    Architecture: Extensible creditPromotions[] array on User model, controlled by env vars. Server grants on /load, logs to ApiUsageLog. Chatbot reads promo status via subscription endpoint piggyback. Notice dismiss persisted in DB.

    Tech Stack: Node.js/Express, MongoDB/Mongoose, React/Redux (admin app), vanilla JS Shadow DOM (chatbot widget)

    Task 4: Server — credit promo logic module¶

    Files: - Create: server/lib/credit-promo.js

    Step 1: Create the module

    const mongoose = require('mongoose');
const User = mongoose.model('User');
const ApiUsageLog = require('../models/ApiUsageLog');
const logger = require('./logger');

const PROMO_CODE = (process.env.CREDIT_PROMO_CODE || '').trim();
const PROMO_CENTS = parseInt(process.env.CREDIT_PROMO_CENTS || '0', 10);

/**
 * Attempt to grant the active credit promo to a user (idempotent).
 * Returns { awarded: true, code, amountCents } if granted, or { awarded: false }.
 */
async function tryGrantPromo(user) {
    if (!PROMO_CODE || PROMO_CENTS <= 0) return { awarded: false };

    // Already claimed this promo?
    if (user.creditPromotions?.some(p => p.code === PROMO_CODE)) {
        return { awarded: false };
    }

    const balanceBefore = user.creditBalance || 0;

    const updated = await User.findOneAndUpdate(
        {
            _id: user._id,
            'creditPromotions.code': { $ne: PROMO_CODE },
        },
        {
            $push: {
                creditPromotions: {
                    code: PROMO_CODE,
                    amountCents: PROMO_CENTS,
                    claimedAt: new Date(),
                    noticeDismissedAt: null,
                },
            },
            $inc: { creditBalance: PROMO_CENTS },
        },
        { new: true }
    );

    if (!updated) return { awarded: false };

    // Fire-and-forget log
    const storeHash = user.context?.replace('stores/', '') || '';
    ApiUsageLog.create({
        storeHash,
        endpoint: 'admin-credit-adjustment',
        subscriptionEvent: 'credit_promo',
        amountCents: PROMO_CENTS,
        balanceBeforeCents: balanceBefore,
        balanceAfterCents: updated.creditBalance,
    }).catch(err => logger.error('CreditPromo.log', err));

    return { awarded: true, code: PROMO_CODE, amountCents: PROMO_CENTS };
}

/**
 * Get the active promo status for chatbot notice.
 * Returns { code, amountCents, noticeDismissed } or null.
 */
function getActivePromo(user) {
    if (!PROMO_CODE || PROMO_CENTS <= 0) return null;

    const entry = user.creditPromotions?.find(p => p.code === PROMO_CODE);
    if (!entry) return null;

    return {
        code: entry.code,
        amountCents: entry.amountCents,
        noticeDismissed: entry.noticeDismissedAt != null,
    };
}

/**
 * Dismiss the promo notice for a user (set noticeDismissedAt).
 */
async function dismissPromoNotice(storeHash, code) {
    await User.findOneAndUpdate(
        { context: `stores/${storeHash}`, 'creditPromotions.code': code },
        { $set: { 'creditPromotions.$.noticeDismissedAt': new Date() } }
    );
}

module.exports = { tryGrantPromo, getActivePromo, dismissPromoNotice };

        await dismissPromoNotice(res.locals.storeHash, code);
