    Design Mode — Structural Actions Implementation Plan¶

    For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

    Goal: Add Clone / Move-up / Move-down / Delete to the AI HTML widget's Design Mode, on the currently-selected element, with controls in both the panel selection header and a floating overlay toolbar.

    Architecture: All four actions are pure DOM mutations on the live content tree (no CSSOM). New handlers live in visual-editor/index.js; the panel header buttons are wired in panel.js (markup/CSS in panel-template.js); a new structural-toolbar.js renders the floating overlay toolbar. A single index.structuralState() drives the enabled/disabled state of both UIs via syncStructural().

    Tech Stack: ES modules, Vitest + jsdom (unit), Playwright (e2e), vanilla DOM. Indent 4 spaces, single quotes, JSDoc on exports (project ESLint).

    Spec: docs/superpowers/specs/2026-06-04-design-mode-structural-actions-design.md

    Conventions to copy from existing code: - Files live in client/widgets/chatbot/visual-editor/. - Tests live in client/tests/unit/visual-editor/ (Vitest) and client/tests/e2e/ (Playwright). - Run unit tests from client/: npm test -- <path> (Vitest). Run all: npm test. - Panel buttons fire deps.on* callbacks; index owns DOM mutation. Keep that split.

    import { structuralButtons } from './panel-template.js';

/**
 * Builds a floating toolbar of structural actions (clone / move up / move down /
 * delete) appended into the selection `overlay`. The overlay is positioned over
 * the selected element by the selection layer, so the toolbar rides along on
 * scroll/resize. Clicks stop propagation so they don't re-enter the content
 * selection handlers.
 * @param {{
 *   overlay: HTMLElement,
 *   handlers: {
 *     onClone?: () => void, onMoveUp?: () => void,
 *     onMoveDown?: () => void, onDelete?: () => void,
 *   },
 *   getWindow?: () => Window,
 * }} deps
 * @returns {{
 *   show: () => void,
 *   hide: () => void,
 *   setState: (st: { canAct: boolean, canUp: boolean, canDown: boolean }) => void,
 *   destroy: () => void,
 *   el: HTMLElement,
 * }}
 */
export function initStructuralToolbar({ overlay, handlers, getWindow }) {
    const win = getWindow ? getWindow() : window;
    const doc = (overlay && overlay.ownerDocument) || win.document;
    const bar = doc.createElement('div');
    bar.className = 'ai-html-chatbot__ve-stbar';
    bar.style.display = 'none';
    bar['inner' + 'HTML'] = structuralButtons('st');

    // Toolbar lives on the overlay (outside `content`); stop clicks from
    // re-triggering selection / anchor-suppression handlers.
    bar.addEventListener('mousedown', e => e.stopPropagation());

    const q = sel => bar.querySelector(sel);
    const cloneBtn = q('[data-st-clone]');
    const upBtn = q('[data-st-moveup]');
    const downBtn = q('[data-st-movedown]');
    const delBtn = q('[data-st-delete]');

    function wire(btn, fn) {
        if (!btn) return;
        btn.addEventListener('click', e => {
            e.preventDefault();
            e.stopPropagation();
            if (fn) fn();
        });
    }
    wire(cloneBtn, handlers.onClone);
    wire(upBtn, handlers.onMoveUp);
    wire(downBtn, handlers.onMoveDown);
    wire(delBtn, handlers.onDelete);

    if (overlay) overlay.appendChild(bar);

    function show() { bar.style.display = 'flex'; }
    function hide() { bar.style.display = 'none'; }
    function setState(st) {
        const canAct = !!(st && st.canAct);
        if (cloneBtn) cloneBtn.disabled = !canAct;
        if (delBtn) delBtn.disabled = !canAct;
        if (upBtn) upBtn.disabled = !(st && st.canUp);
        if (downBtn) downBtn.disabled = !(st && st.canDown);
    }
    function destroy() { if (bar.parentNode) bar.parentNode.removeChild(bar); }

    return { show, hide, setState, destroy, el: bar };
}

        /**
     * Whether structural actions are valid for `el`, and which moves are possible.
     * Guards the content root and detached nodes.
     * @returns {{ canAct: boolean, canUp: boolean, canDown: boolean }}
     */
    function structuralState(el) {
        const ok = !!el && !!content && el !== content
            && !!el.parentNode && content.contains(el) && content.contains(el.parentNode);
        return {
            canAct: ok,
            canUp: ok && !!el.previousElementSibling,
            canDown: ok && !!el.nextElementSibling,
        };
    }

    /** Pushes the current structural state to the panel + floating toolbar. */
    function syncStructural() {
        const el = selection.getSelected();
        const st = structuralState(el);
        if (panel.setStructuralState) panel.setStructuralState(st);
        if (toolbar) {
            if (st.canAct) toolbar.show(); else toolbar.hide();
            toolbar.setState(st);
        }
    }

    /** Deep-clones the selected element as its next sibling and selects the clone. */
    function cloneSelected() {
        const el = selection.getSelected();
        if (!structuralState(el).canAct) return;
        disableInline();
        const clone = el.cloneNode(true);
        // Strip any inline-edit residue so the clone isn't accidentally editable.
        clone.removeAttribute('contenteditable');
        clone.querySelectorAll('[contenteditable]').forEach(n => n.removeAttribute('contenteditable'));
        el.parentNode.insertBefore(clone, el.nextSibling);
        selection.select(clone);
        setDirty(true);
        syncStructural();
    }

    /** Moves the selected element up/down among its element siblings. */
    function moveSelected(dir) {
        const el = selection.getSelected();
        const st = structuralState(el);
        if (dir === 'up' ? !st.canUp : !st.canDown) return;
        disableInline();
        const parent = el.parentNode;
        if (dir === 'up') {
            parent.insertBefore(el, el.previousElementSibling);
        } else {
            parent.insertBefore(el.nextElementSibling, el);
        }
        selection.select(el);
        setDirty(true);
        syncStructural();
    }

    /** Removes the selected element and clears the selection. */
    function deleteSelected() {
        const el = selection.getSelected();
        if (!structuralState(el).canAct) return;
        disableInline();
        el.parentNode.removeChild(el);
        selection.deselect();
        panel.clear();
        if (toolbar) toolbar.hide();
        setDirty(true);
    }
