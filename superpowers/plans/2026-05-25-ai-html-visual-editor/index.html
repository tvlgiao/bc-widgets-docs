    AI HTML Widget — Visual Editor Implementation Plan¶

    For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

    Goal: Add a WYSIWYG, class-based visual style editor to the AI HTML Generator | PapaThemes widget so merchants can edit text/spacing/color/typography/border/alignment per breakpoint inside Page Builder, with live edits committed only on "Save changes".

    Architecture: A "Design mode" added to the existing chatbot Shadow-DOM sidebar. Selecting an element resolves its UUID-prefixed class; edits mutate the matching CSSOM rule inside the live <style> (base for Desktop, @media for Tablet/Mobile) — no sendWidgetUpdate per edit. On Save, the content markup is serialized once and persisted through the existing widget pipeline. Active breakpoint auto-follows the live viewport width.

    Tech Stack: Vanilla ES6 modules (client widgets), Vite build, Vitest + jsdom for unit tests, Playwright for E2E. No framework in the storefront widget.

    Spec: docs/superpowers/specs/2026-05-25-ai-html-visual-editor-design.md Mockup (reference for panel markup/styles): docs/superpowers/specs/2026-05-25-ai-html-visual-editor-mockup.html

    Repo constraints (read before coding)¶

    • Write-hook blocks the literal token inner+HTML. When Claude authors files, that exact token in source is rejected. Use the helpers readInner(el) / writeInner(el, html) defined in Task 6 (dom-html.js) which access the property via bracket notation. Never type the literal token in any new file.
    • Coding style: 4-space indent, single quotes, import/require at top of file, brace on same line.
    • Editor runs only in Page Builder editor mode (isEditorMode), same gating as the chatbot.
    • AI output contract (server/lib/ai-system-prompt.js): every element has a UUID-prefixed class, styles live in a <style> block, no inline styles.

    // Desktop-down. maxWidth is the inclusive upper bound of the breakpoint.
export const BREAKPOINTS = [
    { id: 'desktop', maxWidth: Infinity, mediaMax: null },
    { id: 'tablet', maxWidth: 1024, mediaMax: 1024 },
    { id: 'mobile', maxWidth: 768, mediaMax: 768 },
];

export function breakpointForWidth(width) {
    if (width <= 768) return 'mobile';
    if (width <= 1024) return 'tablet';
    return 'desktop';
}

export function mediaQueryFor(id) {
    const bp = BREAKPOINTS.find(b => b.id === id);
    if (!bp || bp.mediaMax == null) return null;
    return `(max-width: ${bp.mediaMax}px)`;
}

// Watches the live viewport width and reports the active breakpoint.
// onChange(bpId) fires whenever the active breakpoint changes.
export function initBreakpointWatcher(win, onChange) {
    let active = breakpointForWidth(win.innerWidth);
    let timer = null;

    const recompute = () => {
        const next = breakpointForWidth(win.innerWidth);
        if (next !== active) {
            active = next;
            onChange(active);
        }
    };
    const debounced = () => {
        if (timer) win.clearTimeout(timer);
        timer = win.setTimeout(recompute, 100);
    };

    win.addEventListener('resize', debounced);
    const mqls = [768, 1024].map(px => {
        const mql = win.matchMedia(`(max-width: ${px}px)`);
        mql.addEventListener('change', recompute);
        return mql;
    });

    return {
        getActive: () => active,
        destroy() {
            win.removeEventListener('resize', debounced);
            mqls.forEach((mql, i) => mql.removeEventListener('change', recompute));
            if (timer) win.clearTimeout(timer);
        },
    };
}

    const timers = [];
    return {
        innerWidth: width,
        addEventListener: (t, cb) => { (listeners[t] = listeners[t] || []).push(cb); },
        removeEventListener: () => {},
        matchMedia: () => ({ addEventListener() {}, removeEventListener() {} }),
        setTimeout: (cb) => { timers.push(cb); return timers.length; },
        clearTimeout: () => {},
        _fire(t) { (listeners[t] || []).forEach(cb => cb()); },
        _runTimers() { timers.splice(0).forEach(cb => cb()); },
    };
}

    import { BREAKPOINTS } from './breakpoints.js';

// Order of breakpoints that apply at a given active breakpoint (desktop-down).
const LAYERS = {
    desktop: ['desktop'],
    tablet: ['desktop', 'tablet'],
    mobile: ['desktop', 'tablet', 'mobile'],
};

export function elementClass(el) {
    if (el.dataset && el.dataset.cls) return el.dataset.cls;
    return el.classList[0] || null;
}

export function nearestClassedAncestor(el, root) {
    let node = el;
    while (node && node !== root.parentNode) {
        if (node.nodeType === 1 && node.classList && node.classList.length) return node;
        node = node.parentNode;
    }
    return null;
}

// Specificity: [ids, classes, types] → comparable number. Generated code uses
// class selectors only, so a count of class/attr/pseudo tokens is sufficient.
function specificity(selectorText) {
    const ids = (selectorText.match(/#[\w-]+/g) || []).length;
    const classes = (selectorText.match(/\.[\w-]+|\[[^\]]+\]|:[\w-]+/g) || []).length;
    const types = (selectorText.replace(/[#.:\[][^\s>+~]*/g, ' ').match(/[a-z][\w-]*/gi) || []).length;
    return ids * 10000 + classes * 100 + types;
}

function breakpointOfRule(mediaText) {
    if (!mediaText) return 'desktop';
    const m = mediaText.match(/max-width:\s*(\d+)px/);
    if (!m) return 'desktop';
    const px = parseInt(m[1], 10);
    const bp = BREAKPOINTS.find(b => b.mediaMax === px);
    return bp ? bp.id : 'desktop';
}

// index: array of { selectorText, declarations: CSSStyleDeclaration, bp, specificity, order, rule, styleEl }
export function buildRuleIndex(styleEls) {
    const index = [];
    let order = 0;
    for (const styleEl of styleEls) {
        let rules;
        try { rules = styleEl.sheet ? styleEl.sheet.cssRules : []; } catch (e) { rules = []; }
        for (const rule of rules) {
            if (rule.type === 1 /* STYLE_RULE */) {
                pushRule(index, rule, null, styleEl, order++);
            } else if (rule.type === 4 /* MEDIA_RULE */) {
                for (const inner of rule.cssRules) {
                    if (inner.type === 1) pushRule(index, inner, rule.media.mediaText, styleEl, order++);
                }
            }
        }
    }
    return index;
}

function pushRule(index, rule, mediaText, styleEl, order) {
    rule.selectorText.split(',').forEach(sel => {
        const selectorText = sel.trim();
        index.push({
            selectorText,
            declarations: rule.style,
            bp: breakpointOfRule(mediaText),
            specificity: specificity(selectorText),
            order,
            rule,
            styleEl,
        });
    });
}

// Effective value of `prop` for `el` at breakpoint `bp` (desktop-down cascade).
export function computeValue(index, el, prop, bp) {
    const layers = LAYERS[bp] || ['desktop'];
    let winner = null;
    for (const entry of index) {
        if (!layers.includes(entry.bp)) continue;
        let matches;
        try { matches = el.matches(entry.selectorText); } catch (e) { matches = false; }
        if (!matches) continue;
        const value = entry.declarations.getPropertyValue(prop);
        if (!value) continue;
        const important = entry.declarations.getPropertyPriority(prop) === 'important';
        const rank = layerRank(layers, entry.bp);
        const cand = { value, important, specificity: entry.specificity, order: entry.order, rank };
        if (!winner || beats(cand, winner)) winner = cand;
    }
    return winner ? winner.value : '';
}

function layerRank(layers, bp) { return layers.indexOf(bp); }

function beats(a, b) {
    if (a.important !== b.important) return a.important;
    if (a.rank !== b.rank) return a.rank > b.rank;          // later breakpoint layer wins
    if (a.specificity !== b.specificity) return a.specificity > b.specificity;
    return a.order >= b.order;                               // later source order wins
}

export function affectedCount(contentEl, selector) {
    try { return contentEl.querySelectorAll(selector).length; } catch (e) { return 0; }
}

    import { mediaQueryFor } from './breakpoints.js';

// Find (or create) the CSSStyleRule for `.className` at breakpoint `bp`
// inside `styleEl`. For Desktop → a top-level rule; otherwise → a rule nested
// in the matching @media block.
export function getOrCreateRule(styleEl, className, bp) {
    const sheet = styleEl.sheet;
    const selector = '.' + className;
    const media = mediaQueryFor(bp);

    if (!media) {
        const found = findStyleRule(sheet, selector);
        if (found) return found;
        const i = sheet.insertRule(`${selector} {}`, sheet.cssRules.length);
        return sheet.cssRules[i];
    }

    let mediaRule = findMediaRule(sheet, media);
    if (!mediaRule) {
        const i = sheet.insertRule(`@media ${media} {}`, sheet.cssRules.length);
        mediaRule = sheet.cssRules[i];
    }
    const found = findStyleRule(mediaRule, selector);
    if (found) return found;
    const j = mediaRule.insertRule(`${selector} {}`, mediaRule.cssRules.length);
    return mediaRule.cssRules[j];
}

function findStyleRule(container, selector) {
    for (const r of container.cssRules) {
        if (r.type === 1 && r.selectorText === selector) return r;
    }
    return null;
}

function findMediaRule(sheet, media) {
    const norm = media.replace(/\s+/g, '');
    for (const r of sheet.cssRules) {
        if (r.type === 4 && r.media.mediaText.replace(/\s+/g, '') === norm) return r;
    }
    return null;
}

// Apply one property change to the live stylesheet. Returns the edited rule.
export function applyEdit(styleEl, el, className, cssProp, value, bp) {
    const rule = getOrCreateRule(styleEl, className, bp);
    if (value === '' || value == null) {
        rule.style.removeProperty(cssProp);
    } else {
        rule.style.setProperty(cssProp, value);
    }
    styleEl.dataset.ptEdited = '1';
    return rule;
}

    import { nearestClassedAncestor } from './css-resolver.js';

export function initSelection({ stage, content, overlay, hoverOverlay, onSelect, onDeselect }) {
    let current = null;

    function position(ov, el) {
        const s = stage.getBoundingClientRect();
        const b = el.getBoundingClientRect();
        ov.style.display = 'block';
        ov.style.left = (b.left - s.left) + 'px';
        ov.style.top = (b.top - s.top) + 'px';
        ov.style.width = b.width + 'px';
        ov.style.height = b.height + 'px';
    }

    function onClick(e) {
        const el = nearestClassedAncestor(e.target, content);
        if (!el || !content.contains(el)) return;
        e.preventDefault();
        e.stopPropagation();
        current = el;
        position(overlay, el);
        onSelect(el);
    }

    function onMove(e) {
        const el = nearestClassedAncestor(e.target, content);
        if (el && el !== current && content.contains(el)) position(hoverOverlay, el);
        else hoverOverlay.style.display = 'none';
    }

    function onKey(e) {
        if (e.key === 'Escape') deselect();
        if (e.key === 'ArrowUp' && current) {
            const parent = nearestClassedAncestor(current.parentNode, content);
            if (parent) { current = parent; position(overlay, parent); onSelect(parent); }
        }
    }

    function deselect() {
        current = null;
        overlay.style.display = 'none';
        if (onDeselect) onDeselect();
    }

    function reposition() { if (current) position(overlay, current); }

    content.addEventListener('click', onClick, true);
    content.addEventListener('mousemove', onMove);
    content.addEventListener('mouseleave', () => { hoverOverlay.style.display = 'none'; });
    document.addEventListener('keydown', onKey);
    window.addEventListener('scroll', reposition, true);
    window.addEventListener('resize', reposition);

    return {
        getSelected: () => current,
        reposition,
        deselect,
        destroy() {
            content.removeEventListener('click', onClick, true);
            content.removeEventListener('mousemove', onMove);
            document.removeEventListener('keydown', onKey);
            window.removeEventListener('scroll', reposition, true);
            window.removeEventListener('resize', reposition);
        },
    };
}

    import { elementClass, affectedCount, computeValue } from './css-resolver.js';

export const pxToNumber = v => Math.round(parseFloat(v) || 0);

export function rgbToHex(rgb) {
    const m = (rgb || '').match(/\d+/g);
    if (!m) return '#000000';
    return '#' + m.slice(0, 3).map(x => (+x).toString(16).padStart(2, '0')).join('');
}

const PX_CONTROLS = new Set([
    'padding-top', 'padding-right', 'padding-bottom', 'padding-left',
    'margin-top', 'margin-right', 'margin-bottom', 'margin-left',
    'font-size', 'border-width', 'border-radius', 'letter-spacing',
]);

export function cssPropForControl(id) {
    if (id === 'line-height-u') return 'line-height';
    return id;
}

// Build the live panel controller.
// deps: { root, getActiveBp, onEdit(el, className, cssProp, value, bp), onSave, onDiscard }
export function initPanel(deps) {
    const { root } = deps;
    const q = sel => root.querySelector(sel);
    const empty = q('[data-ve-empty]');
    const edit = q('[data-ve-edit]');
    const tagEl = q('[data-ve-tag]');
    const classEl = q('[data-ve-class]');
    const affectsEl = q('[data-ve-affects]');
    const noteEl = q('[data-ve-note]');
    let selected = null;
    let index = null;

    function setIndex(ruleIndex) { index = ruleIndex; }

    function show(el, contentEl) {
        selected = el;
        const cls = elementClass(el);
        const bp = deps.getActiveBp();
        tagEl.textContent = el.tagName.toLowerCase();
        classEl.textContent = '.' + cls;
        const n = affectedCount(contentEl, '.' + cls);
        affectsEl.hidden = n <= 1;
        affectsEl.dataset.count = n;
        noteEl.dataset.target = mediaQueryFor(bp) || 'base';
        empty.hidden = true;
        edit.hidden = false;
        refresh();
        markActiveBp(bp);
    }

    function refresh() {
        if (!selected) return;
        const cs = getComputedStyle(selected);
        root.querySelectorAll('[data-p]').forEach(input => {
            const id = input.dataset.p;
            const prop = cssPropForControl(id);
            if (PX_CONTROLS.has(id)) { input.value = pxToNumber(cs.getPropertyValue(prop)); return; }
            if (id === 'color') { input.value = rgbToHex(cs.color); return; }
            if (id === 'background-color') { input.value = rgbToHex(cs.backgroundColor); return; }
            if (id === 'border-color') { input.value = rgbToHex(cs.borderTopColor); return; }
            if (id === 'border-style') { input.value = cs.borderTopStyle || 'none'; return; }
            if (id === 'font-weight') { input.value = String(cs.fontWeight) || '400'; return; }
            if (id === 'line-height-u') {
                const lh = cs.lineHeight === 'normal' ? 1.4 : parseFloat(cs.lineHeight) / parseFloat(cs.fontSize);
                input.value = Math.round(lh * 10) / 10; return;
            }
            input.value = cs.getPropertyValue(prop);
        });
    }

    function markActiveBp(bp) {
        root.querySelectorAll('[data-ve-bprow] button').forEach(b =>
            b.classList.toggle('active', b.dataset.bp === bp));
    }

    function valueForControl(input) {
        const id = input.dataset.p;
        if (PX_CONTROLS.has(id)) return (parseFloat(input.value) || 0) + 'px';
        return input.value;
    }

    root.querySelectorAll('[data-p]').forEach(input => {
        input.addEventListener('input', () => {
            if (!selected) return;
            deps.onEdit(selected, elementClass(selected), cssPropForControl(input.dataset.p),
                valueForControl(input), deps.getActiveBp());
        });
    });

    q('[data-ve-save]').addEventListener('click', () => deps.onSave());
    q('[data-ve-discard]').addEventListener('click', () => deps.onDiscard());

    function clear() { selected = null; edit.hidden = true; empty.hidden = false; }
    function onBreakpointChange(bp) { markActiveBp(bp); noteEl.dataset.target = mediaQueryFor(bp) || 'base'; refresh(); }

    return { show, refresh, clear, setIndex, onBreakpointChange, getSelected: () => selected };
}

export function initVisualEditor(deps) {

    let dirty = false;

    const styleEls = () => Array.from(deps.content.querySelectorAll('style'));
    let ruleIndex = buildRuleIndex(styleEls());

    });

        root: deps.panelRoot,
        getActiveBp: () => watcher.getActive(),
        onEdit: (el, className, cssProp, value, bp) => {
            const target = el.closest('style') ? null : styleEls()[0];
            const styleEl = chooseStyleEl(el);
            applyEdit(styleEl, el, className, cssProp, value, bp);
            selection.reposition();
            setDirty(true);
        },

        onDiscard: discard,
    });

    const selection = initSelection({
        stage: deps.stage,
        content: deps.content,
        overlay: deps.overlay,
        hoverOverlay: deps.hoverOverlay,
        onSelect: el => { panel.setIndex(ruleIndex); panel.show(el, deps.content); },
        onDeselect: () => panel.clear(),
    });

    // Choose the <style> that owns the element's class rules; default to the first.
    function chooseStyleEl(el) {
        const els = styleEls();
        return els[0] || createOwnedStyle();
    }
    function createOwnedStyle() {
        const s = document.createElement('style');
        s.dataset.ptOwned = '1';
        deps.content.insertBefore(s, deps.content.firstChild);
        ruleIndex = buildRuleIndex(styleEls());
        return s;
    }

    function setDirty(v) {
        dirty = v;
        const dot = deps.panelRoot.querySelector('[data-ve-dirty]');
        const save = deps.panelRoot.querySelector('[data-ve-save]');
        const discardBtn = deps.panelRoot.querySelector('[data-ve-discard]');
        if (dot) dot.classList.toggle('on', v);
        if (save) save.disabled = !v;
        if (discardBtn) discardBtn.disabled = !v;
        if (deps.onDirtyChange) deps.onDirtyChange(v);
    }

    function save() {
        const html = serializeContent(deps.content);
        deps.persistContent(html);
        ruleIndex = buildRuleIndex(styleEls());
        setDirty(false);
    }

        rebuildIndex,
        deselect: () => { selection.deselect(); panel.clear(); },
        destroy() { watcher.destroy(); selection.destroy(); },

    // @ts-check
const { test, expect } = require('@playwright/test');

const HARNESS = '/tests/fixtures/visual-editor-harness.html';

test('selecting a shared-class card shows the affects-N badge and edits all', async ({ page }) => {
    await page.goto(HARNESS);
    await page.locator('.pt-w-card').first().click();
    await expect(page.locator('[data-ve-affects]')).toBeVisible();
    await expect(page.locator('[data-ve-affects]')).toContainText('3');

    const before = await page.locator('.pt-w-card').nth(2).evaluate(el => getComputedStyle(el).paddingTop);
    await page.locator('[data-p="padding-top"]').fill('60');
    await page.locator('[data-p="padding-top"]').dispatchEvent('input');
    const after = await page.locator('.pt-w-card').nth(2).evaluate(el => getComputedStyle(el).paddingTop);
    expect(after).toBe('60px');
    expect(after).not.toBe(before);
});

test('class-less child selects nearest classed ancestor', async ({ page }) => {
    await page.goto(HARNESS);
    await page.locator('.pt-w-card h3').first().click();
    await expect(page.locator('[data-ve-class]')).toHaveText(/pt-w-card-h|pt-w-card/);
});

test('no save until Save clicked, then exactly one sendWidgetUpdate', async ({ page }) => {
    await page.goto(HARNESS);
    await page.locator('.pt-w-card').first().click();
    await page.locator('[data-p="padding-top"]').fill('40');
    await page.locator('[data-p="padding-top"]').dispatchEvent('input');
    await page.locator('[data-p="padding-bottom"]').fill('40');
    await page.locator('[data-p="padding-bottom"]').dispatchEvent('input');
    expect(await page.evaluate(() => window.__saves.length)).toBe(0);
    await expect(page.locator('[data-ve-save]')).toBeEnabled();
    await page.locator('[data-ve-save]').click();
    expect(await page.evaluate(() => window.__saves.length)).toBe(1);
    await expect(page.locator('[data-ve-save]')).toBeDisabled();
});

test('editing at mobile creates a @media rule and leaves desktop unchanged', async ({ page }) => {
    await page.goto(HARNESS);
    await page.setViewportSize({ width: 375, height: 800 });
    await page.locator('.pt-w-card').first().click();
    await page.locator('[data-p="padding-top"]').fill('8');
    await page.locator('[data-p="padding-top"]').dispatchEvent('input');
    await page.locator('[data-ve-save]').click();
    const saved = await page.evaluate(() => window.__saves[0].content);
    expect(saved).toContain('max-width: 768px');
});
